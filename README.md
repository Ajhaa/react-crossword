## Install

```bash
npm install --save react-crossword

or

yarn add react-crossword
```

## Usage

```jsx
import React from 'react';

import { Demo } from 'react-crossword';

const Example = ({ message }) => <Demo type={'warning'}>{message}</Demo>;

export default Example;
```
